% Centred figure command
\newcommand{\cfig}[3]{
\begin{figure}[H]
	\centering
	\includegraphics[width=#3\linewidth]{Images/#1}
    \caption{#2}
	\label{fig:#1}
\end{figure}
}

% Figure on right side
\newcommand{\rfig}[3]{
\begin{wrapfigure}{r}{#3\linewidth}
	\centering
	\includegraphics[width=\linewidth]{Images/#1}
	\caption{#2}
	\label{fig:#1}
	\vspace*{-13pt}
\end{wrapfigure}
}

% Two figures side by side
\newcommand{\twofig}[5]{
\begin{figure}[H]
	\begin{minipage}{0.48\linewidth}
		\centering
		\includegraphics[width=#5\linewidth]{Images/#1}
		\caption{#2}
		\label{fig:#1}
	\end{minipage}\hfill
	\begin{minipage}{0.48\linewidth}
		\centering
		\includegraphics[width=#5\linewidth]{Images/#3}
		\caption{#4}
		\label{fig:#3}
	\end{minipage}
\end{figure}
}

% Create bibliography page
\newcommand{\bibliopage}{
	\phantomsection
	\printbibliography[heading=bibintoc,title=Bibliography]
}
